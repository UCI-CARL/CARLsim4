61 int SpikeGeneratorFromVector::nextSpikeTime(CARLsim* sim, int grpId, int nid, int currentTime, int lastScheduledSpikeTime, int endOfTimeSlice) {
62 
63  // schedule spike if vector index valid and spike within scheduling time slice
64  if (currentIndex_ < size_ && spkTimes_[currentIndex_] < endOfTimeSlice) {
65  return spkTimes_[currentIndex_++];
66  }
67 
68  return -1; // -1: large positive number
69 }

71 void SpikeGeneratorFromVector::checkSpikeVector() {
72  UserErrors::assertTrue(size_>0,UserErrors::CANNOT_BE_ZERO, "SpikeGeneratorFromVector", "Vector size");
73  for (int i=0; i<size_; i++) {
74  std::stringstream var; var << "spkTimes[" << currentIndex_ << "]";
75  UserErrors::assertTrue(spkTimes_[i]>0,UserErrors::CANNOT_BE_ZERO, "SpikeGeneratorFromVector", var.str());
76  }
